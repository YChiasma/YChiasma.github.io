rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    match /users/{uid} {
      allow read: if request.auth != null;
      allow write: if request.auth.uid == uid;
    }

    match /users/{uid}/streaks/{streakId} {
      allow read: if resource.data.public == true
                  || request.auth.uid == uid;

      allow write: if request.auth.uid == uid;
    }

    match /users/{uid}/streaks/{streakId}/days/{dayId} {
      allow read: if
        get(/databases/$(database)/documents/users/$(uid)/streaks/$(streakId)).data.public == true
        || request.auth.uid == uid;

      allow write: if request.auth.uid == uid;
    }

  }
}
